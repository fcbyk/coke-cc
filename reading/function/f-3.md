## 3 分析运行结果

观察以下代码，它的运行结果会是怎么样呢？

::: code-group

```cpp [代码1]
#include <iostream>
using namespace std;

void fun1() {
    cout << "函数1" << endl;
}

void fun2() {
    cout << "函数2" << endl;
    fun1();
}

int main() {
    fun2();
    cout << "主函数" << endl;
    return 0;
}
```

```cpp [代码2]
#include <iostream>
using namespace std;

int add(int a,int b) {
    a++;
    return a;
    return a+b;
}

int main() {
    cout << add(7,9);
    return 0;
}
```

```cpp [代码3]
#include <iostream>
using namespace std;

int myMinus(int a = 100, int b = 200) {
    int result = a - b;
    return result;
}

int main() {
    int hhh = myMinus() + myMinus(300,100);
    cout << hhh;
    return 0;
}
```

```cpp [代码4]
#include <iostream>
using namespace std;

int big(int a) {
    return a+10;
}

int main() {
    cout << big(big(5+5));
    return 0;
}
```
